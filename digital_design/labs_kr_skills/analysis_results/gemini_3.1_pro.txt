         ```vhdl  
         use IEEE.std_logic_1164.all;

         entity BRLSHFT8 is
             port (
                 -- Объединяем одиночные порты в векторы для удобства поведенческого описания.

                 I : in  std_logic_vector(0 to 7); 
                 S : in  std_logic_vector(2 downto 0); -- S2, S1, S0
                 O : out std_logic_vector(0 to 7)
             );
         end entity BRLSHFT8;

         architecture Behavioral of BRLSHFT8 is
         begin
             process(I, S)
             begin
                 -- Описание строго по строкам Logic Table:
                 case S is
                     when "000" => 
                         O <= I(0 to 7); -- a b c d e f g h
                     when "001" => 
                         O <= I(1 to 7) & I(0); -- b c d e f g h a
                     when "010" => 
                         O <= I(2 to 7) & I(0 to 1); -- c d e f g h a b
                     when "011" => 
                         O <= I(3 to 7) & I(0 to 2); -- d e f g h a b c
                     when "100" => 
                         O <= I(4 to 7) & I(0 to 3); -- e f g h a b c d
                     when "101" => 
                         O <= I(5 to 7) & I(0 to 4); -- f g h a b c d e
                     when "110" => 
                         O <= I(6 to 7) & I(0 to 5); -- g h a b c d e f
                     when "111" => 
                         O <= I(7) & I(0 to 6); -- h a b c d e f g
                     when others => 
                         O <= (others => '0');
                 end case;
             end process;
         end architecture Behavioral;
         ```
